XPlot

Plotly Line and Scatter Plots

Basic Line Plot

open XPlot.Plotly

let trace1 =
    Scatter(
        x = [1; 2; 3; 4],
        y = [10; 15; 13; 17]
    )

let trace2 =
    Scatter(
        x = [2; 3; 4; 5],
        y = [16; 5; 11; 9]
    )

let chart1 =
    [trace1; trace2]
    |> Chart.Plot
    |> Chart.WithWidth 700
    |> Chart.WithHeight 500

Line and Scatter Plot

let lineTrace1 =
    Scatter(
        x = [1; 2; 3; 4],
        y = [10; 15; 13; 17],
        mode = "markers"
    )

    Scatter(

    )

let lineTrace3 =

        x = [1; 2; 3; 4],
        y = [12; 9; 15; 12],
        mode = "lines+markers"
    )

let chart2 =
    [lineTrace1; lineTrace2; lineTrace3]
    |> Chart.Plot
    |> Chart.WithWidth 700
    |> Chart.WithHeight 500
